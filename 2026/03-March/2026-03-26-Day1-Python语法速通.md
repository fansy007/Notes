### 1.1 变量与数据类型

Python是**动态类型语言**，不需要声明类型。

```python
# Java: int a = 10;
a = 10                    # int自动推断

# Java: String name = "haining";
name = "haining"          # str

# Java: double pi = 3.14;
pi = 3.14                # float

# 多个赋值
x, y, z = 1, 2, 3

# 类型查看
print(type(a))            # <class 'int'>
print(type(name))        # <class 'str'>
```

**对比Java：**
- 不需要声明类型
- 不需要分号结尾
- 不需要 `int`、`String` 等关键字

### 1.7 列表推导式（重要！）

### 2.1 函数

```python
# 基本语法
def greet(name):
    """这是函数的文档字符串"""
    return f"Hello, {name}!"

# 调用
message = greet("haining")
print(message)  # Hello, haining!

# 默认参数（类似Java）
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

print(greet("haining"))           # Hello, haining!
print(greet("haining", "Hi"))     # Hi, haining!

# 可变参数
def sum_all(*args):
    """*args接收任意数量的参数"""
    total = 0
    for num in args:
        total += num
    return total

print(sum_all(1, 2, 3))        # 6
print(sum_all(1, 2, 3, 4, 5)) # 15

# 关键字参数
def show_info(name, age=18, city="Shanghai"):
    return f"{name}, {age}, {city}"

print(show_info("haining"))                    # 使用默认值
print(show_info("haining", age=43))          # 指定部分
print(show_info(name="haining", city="Beijing", age=43))  # 指定全部

# **kwargs 接收关键字参数
def print_all(**kwargs):
    for key, value in kwargs.items():
        print(f"{key}: {value}")

print_all(name="haining", job="量化", city="Shanghai")
# name: haining
# job: 量化
# city: Shanghai
```

```java

    return "Hello, " + name + "!";
}
```

### 2.2 类（Class）

```python
class Person:
    """人类"""
    
    # 类变量（所有实例共享）
    species = "Human"
    
    # 构造方法
    def __init__(self, name, age=18):
        # 实例变量
        self.name = name
        self.age = age
    
    # 实例方法
    def greet(self):
        return f"Hi, I'm {self.name}"
    
    # 私有方法（下划线约定）
    def _private_method(self):
        print("这是私有的")
    
    # 字符串表示
    def __str__(self):
        return f"Person({self.name}, {self.age})"
    
    # 类似Java的toString
    def __repr__(self):
        return self.__str__()

# 创建实例
p = Person("haining", 43)
print(p.name)        # haining
print(p.greet())     # Hi, I'm haining
print(p.species)     # Human
print(str(p))        # Person(haining, 43)

# 继承
class Stock:
    def __init__(self, code, price):
        self.code = code
        self.price = price
    
    def get_value(self, shares):
        return self.price * shares

class IndexFund(Stock):
    """指数基金，继承自Stock"""
    
    def __init__(self, code, price, name):
        super().__init__(code, price)  # 调用父类构造
        self.name = name
    
    def get_info(self):
        return f"{self.name}({self.code}): {self.price}"

# 多态
fund = IndexFund("399006", 2000, "创业板指数")
print(fund.get_value(100))  # 200000
print(fund.get_info())      # 创业板指数(399006): 2000
```

**Java对比：**

| Java | Python |
|------|--------|
| `public class Person` | `class Person:` |
| `this.name = name` | `self.name = name` |
| `void method()` | `def method(self):` |
| `super()` | `super()` |
| `public String name` | `self.name` (都是public) |

### 2.3 装饰器（Decorator）— 重要！

装饰器是Python最强大的特性之一，类似Java的AOP。

```python
# 场景：给函数添加计时功能

# Java的做法：切面编程或者手动包装
# Python的做法：装饰器

import time

# 定义装饰器
def timer(func):
    """计时装饰器"""
    def wrapper(*args, **kwargs):
        start = time.time()
        result = func(*args, **kwargs)
        end = time.time()
        print(f"{func.__name__} 执行时间: {end - start:.4f}秒")
        return result
    return wrapper

# 使用装饰器（@符号）
@timer
def slow_function():
    time.sleep(1)
    return "Done"

# 等同于：slow_function = timer(slow_function)

result = slow_function()
# slow_function 执行时间: 1.0000秒

# 带参数的装饰器
def repeat(times):
    """重复执行指定次数"""
    def decorator(func):
        def wrapper(*args, **kwargs):
            for _ in range(times):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(3)
def say_hello():
    print("Hello!")

say_hello()
# Hello!
# Hello!
# Hello!
```

**应用场景（量化回测代码中）：**

```python
# 你会看到的装饰器

@timer                         # 计时

    # ...
```

### 2.4 上下文管理器（with）— 重要！

类似Java的try-with-resources，自动管理资源。

```python
# Java的做法
try (FileReader reader = new FileReader("file.txt")) {
    // 读取文件
} // 自动关闭

# Python的做法
with open("file.txt", "r") as f:
    content = f.read()
# 文件自动关闭

# 读取CSV文件
import pandas as pd
with open("stock_data.csv", "r") as f:
    df = pd.read_csv(f)
    print(df.head())

# 或者pandas直接读
df = pd.read_csv("stock_data.csv")
print(df.head())

# 写入文件
with open("output.txt", "w") as f:
    f.write("Hello World")

# 自定义上下文管理器
class FileManager:
    def __init__(self, filename, mode):
        self.filename = filename
        self.mode = mode
        self.file = None
    
    def __enter__(self):
        self.file = open(self.filename, self.mode)
        return self.file
    
    def __exit__(self, exc_type, exc_val, exc_tb):
        if self.file:
            self.file.close()
        # 返回True可以抑制异常
        return False

with FileManager("test.txt", "w") as f:
    f.write("Hello!")

# 使用contextlib简化
from contextlib import contextmanager

@contextmanager
def timer_context(name):
    start = time.time()
    yield name
    end = time.time()
    print(f"{name} 执行了 {end - start:.2f}秒")

with timer_context("任务1"):
    time.sleep(1)
```

### 3.3 实用技巧

```python
# 条件筛选
prices = np.array([100, 200, 150, 80, 300])

# 找出价格大于150的
print(prices[prices > 150])  # [200 300]

# 找出价格大于150的索引
print(np.where(prices > 150))  # (array([1, 4]),)

# 归一化
normalized = (prices - prices.mean()) / prices.std()
print(normalized)

# 计算收益率
returns = np.diff(prices) / prices[:-1] * 100
print(returns)  # 百分比收益率

# 累计收益
cumulative = (1 + returns / 100).cumprod()
print(cumulative)  # 累计收益倍数
```
