### 1.2 DataFrame — 核心数据结构

DataFrame = 二维表格，类似Excel表格或SQL表。

```python
# 创建DataFrame
data = {
    'stock_code': ['300001', '300002', '300003', '300004'],
    'name': ['宁德时代', '亿纬锂能', '阳光电源', '汇川技术'],
    'price': [500, 80, 120, 60],
    'pe': [30, 25, 40, 35]
}

df = pd.DataFrame(data)
print(df)
#   stock_code   name  price  pe
# 0    300001  宁德时代   500  30
# 1    300002  亿纬锂能    80  25
# 2    300003  阳光电源   120  40
# 3    300004  汇川技术    60  35

# 查看基本信息
print(df.head())      # 前5行
print(df.tail())       # 后5行
print(df.info())       # 列信息
print(df.describe())   # 统计摘要

# 列操作
print(df['name'])              # 获取单列
print(df[['name', 'price']])   # 获取多列

# 新增列
df['market_cap'] = df['price'] * 100000000  # 市值
df['is_expensive'] = df['price'] > 100      # 布尔列
```

### 1.3 数据选择 — 重点！

DataFrame有多种选择数据的方式：

```python
# 准备数据
df = pd.DataFrame({
    'stock_code': ['300001', '300002', '300003', '300004', '300005'],
    'name': ['A', 'B', 'C', 'D', 'E'],
    'price': [500, 80, 120, 60, 200],
    'pe': [30, 25, 40, 35, 28],
    'industry': ['新能源', '新能源', '光伏', '自动化', '新能源']
})

# 1. 按列名选择（推荐）
print(df['price'])           # 单列，Series
print(df[['price', 'pe']])  # 多列，DataFrame

# 2. 按行索引选择 — loc（按标签）
print(df.loc[0])             # 第0行
print(df.loc[0:2])          # 第0到2行（inclusive）
print(df.loc[df['price'] > 100])  # 布尔索引

# 3. 按行号选择 — iloc（按位置）
print(df.iloc[0])           # 第0行
print(df.iloc[1:3])         # 第1到2行（exclusive）
print(df.iloc[:, 1])        # 所有行，第1列

# 4. 组合条件选择
print(df.loc[(df['price'] > 100) & (df['pe'] < 35)])  # AND
print(df.loc[(df['industry'] == '新能源') | (df['price'] > 400)])  # OR
print(df.loc[~df['name'].isin(['A', 'B'])])  # NOT

# 5. query方法（类似SQL）
print(df.query('price > 100 and pe < 35'))
```

**对比Java：**

| Pandas | Java |
|--------|------|
| `df['col']` | `df.get("col")` |
| `df.loc[0]` | `df.getRow(0)` |
| `df.loc[condition]` | `df.filter(row -> condition)` |
| `df.query('...')` | `df.where("...")` |

### 1.4 布尔索引（重要！）

```python

df['price'] > 100          # 返回布尔Series

# PE < 35 且 市值 > 100亿
condition = (df['pe'] < 35) & (df['price'] * 1e8 > 1e10)
```

## 下午：数据清洗 + 筛选

### 2.1 缺失值处理

```python
# 创建有缺失值的数据
df = pd.DataFrame({
    'stock_code': ['300001', '300002', '300003', '300004'],
    'name': ['A', 'B', None, 'D'],  # None表示缺失
    'price': [500, None, 120, 60],
    'pe': [30, 25, None, 35]
})

print(df)
print(df.isnull())           # 哪些是缺失值
print(df.notnull())          # 哪些不是缺失值

# 删除缺失值
df_clean = df.dropna()                    # 删除有缺失值的行
df_clean = df.dropna(subset=['price'])) # 只删除price列缺失的行

# 填充缺失值
df['price'].fillna(df['price'].mean())  # 用均值填充
df['name'].fillna('未知')                # 用字符串填充
df.fillna(0)                            # 全部填充0

# 用前一个/后一个值填充
df['price'].fillna(method='ffill')  # 前值填充
df['price'].fillna(method='bfill')  # 后值填充
```

### 2.3 分组统计（类似SQL GROUP BY）

```python
# 按行业分组
grouped = df.groupby('industry')

# 分组后聚合
grouped['price'].mean()           # 每组平均价
grouped['price'].sum()           # 每组总价
grouped['price'].agg(['mean', 'max', 'min'])  # 多重聚合

# 完整示例
result = df.groupby('industry').agg({
    'price': ['mean', 'max', 'min', 'count'],
    'pe': 'mean'
}).round(2)

print(result)
```

**Java对比：**

```java
// Java Stream
df.groupBy("industry")
  .agg(
    col("price").mean().as("avg_price"),
    col("price").max().as("max_price")
  )
```

### 2.4 数据合并

```python
# 创建两个表
df1 = pd.DataFrame({'stock_code': ['300001', '300002'], 
                     'name': ['A', 'B']})
df2 = pd.DataFrame({'stock_code': ['300001', '300002'], 
                     'price': [500, 80]})

# 合并（类似SQL JOIN）
merged = pd.merge(df1, df2, on='stock_code', how='left')
print(merged)
#   stock_code name  price
# 0    300001    A    500
# 1    300002    B     80

# concat（类似SQL UNION ALL）
df3 = pd.DataFrame({'stock_code': ['300003'], 'name': ['C']})
concatenated = pd.concat([df1, df3], ignore_index=True)
```

### 2.5 实战：选股筛选

```python
# 模拟股票数据
stocks = pd.DataFrame({
    'code': ['300001', '300002', '300003', '300004', '300005',
             '300006', '300007', '300008', '300009', '300010'],
    'name': ['A', 'B', 'C', 'D', 'E', 'F', 'G', 'H', 'I', 'J'],
    'price': [500, 80, 120, 60, 200, 300, 150, 90, 180, 70],
    'pe': [30, 25, 40, 35, 28, 45, 32, 22, 38, 27],
    'roe': [15, 8, 20, 5, 12, 18, 10, 6, 14, 9],
    'profit_growth': [20, -5, 30, -10, 15, 25, 8, -3, 18, 5]
})

# 筛选条件
# 1. PE < 40
condition1 = stocks['pe'] < 40

# 2. ROE > 10%
condition2 = stocks['roe'] > 10

# 3. 净利润增长 > 0
condition3 = stocks['profit_growth'] > 0

# 4. 股价 < 300
condition4 = stocks['price'] < 300

# 组合筛选
selected = stocks[condition1 & condition2 & condition3 & condition4]

# 按ROE排序
selected = selected.sort_values('roe', ascending=False)

print(f"符合条件: {len(selected)} 只股票")
print(selected)
```

## 今日总结

### 核心API

| 功能 | 方法 |
|------|------|
| 创建 | `pd.DataFrame(data)` |
| 选择列 | `df['col']` |
| 选择行 | `df.loc[idx]` / `df.iloc[n]` |
| 布尔筛选 | `df[condition]` |
| 分组 | `df.groupby('col')` |
| 合并 | `pd.merge(df1, df2, on='col')` |
| 排序 | `df.sort_values('col')` |
| 缺失值 | `df.dropna()` / `df.fillna()` |
| 读写CSV | `pd.read_csv()` / `df.to_csv()` |
